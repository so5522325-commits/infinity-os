DYNAMIC FEED LOADER (Home Screen ke liye) async function loadAllPosts() { const feedContainer = document.getElementById('home-feed-container'); if(!feedContainer) return; const dbRef = ref(db, 'users'); feedContainer.innerHTML = ""; get(dbRef).then((snapshot) => { snapshot.forEach((userSnapshot) => { const userData = userSnapshot.val(); if (userData.posts) { Object.values(userData.posts).forEach(post => { renderPostCard(post, feedContainer); }); } }); }); } function renderPostCard(post, container) { const card = document.createElement('div'); card.className = 'video-card'; card.innerHTML = `
FEED
${post.caption}
${post.media ? (post.mediaType === 'video' ? `` : ``) : ""}
`; container.appendChild(card); } // 2. PROFILE POSTS LOADER (Profile Page ke liye) async function loadUserPosts(username) { const postsRef = ref(db, `users/${username}/posts`); const gallery = document.getElementById('user-gallery'); if(!gallery) return; get(postsRef).then((snapshot) => { gallery.innerHTML = ""; if (snapshot.exists()) { snapshot.forEach((child) => { const post = child.val(); const item = document.createElement('div'); item.className = 'gallery-item'; item.innerHTML = `
${post.mediaType === 'video' ? `` : ``}
${post.caption}
`; gallery.appendChild(item); }); } else { gallery.innerHTML = `

Abhi tak koi post nahi ki hai.

`; } }); } // 3. UPDATED PUBLISH FUNCTION window.publishUserPost = async function() { const caption = document.getElementById('post-caption').value.trim(); const activeUser = localStorage.getItem('secure_session_user'); if (!activeUser) return; if (!caption && !tempPostMedia.data) { alert("Post me kuch likhein ya media add karein!"); return; } const postBtn = document.getElementById('submit-post-btn'); postBtn.innerText = "Uploading..."; postBtn.disabled = true; try { const postsRef = ref(db, `users/${activeUser}/posts`); const newPostRef = push(postsRef); await set(newPostRef, { caption: caption, media: tempPostMedia.data || "", mediaType: tempPostMedia.type || "", timestamp: new Date().toISOString() }); // Reset document.getElementById('post-caption').value = ""; document.getElementById('post-media-preview').style.display = 'none'; tempPostMedia = { data: "", type: "" }; // Refresh Feeds loadUserPosts(activeUser); loadAllPosts(); postBtn.innerText = "Post Published!"; setTimeout(() => postBtn.innerText = "Publish Post", 2000); postBtn.disabled = false; } catch (e) { alert("Error uploading post."); postBtn.disabled = false; } };
